
## 1.개발을 하다가 모바일 버튼을 두 번 눌러야 하는 현상 발생

- 버튼은 정상적으로 작동하지만 두 번 눌러야 열리고 두 번 열려야 닫힌다.



----

### 해결방안 

$(document).ready(function () {  
    $('#topBox').on('click', function (event) {  
            $('#listBox').toggle();  
            $(this).toggleClass('pressed');  
            event.stopPropagation();  
        }  
    )})
